pipeline{
	agent any
	stages{
	stage('checkout'){
	steps{
	echo "Cloning repo"
	git url:"https://github.com/rohini-garlapati/Devopslabweek8.git",
	branch:'master'
	}
	}
	stage('Build'){
	steps{
	echo "Build Docker Image"
	bat "docker build -t webapp ."
	}
	}
	stage('Run'){
	steps{
	echo "Run application in Docker Container"
	bat "docker rm -f mycontainer || exit0"
	bat "docker run -d -p 5000:5000 --name mycontainer webapp"
	}
	}
	}
	post{
	success{
	echo 'Pipeline completed successfully!'
	}
	failure{
	echo 'Pipeline failed.Please check the logs'
	}
	
	}
}
